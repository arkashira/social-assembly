# tests/docs/test_troubleshooting_doc.py
import re
from pathlib import Path

DOCS = Path("/opt/axentx/social-assembly/docs")
TROUBLE = DOCS / "troubleshooting.md"
DEPLOY = DOCS / "deployment.md"

REQUIRED_TROUBLESHOOTING_TOPICS = ["database", "reverse proxy", "ssl"]
PLACEHOLDER_PATTERNS = [r"\bTODO\b", r"\bFIXME\b", r"\bXXX\b", r"<your-[^>]+>"]

def test_troubleshooting_file_exists():
    assert TROUBLE.exists(), "docs/troubleshooting.md must exist"
    assert TROUBLE.stat().st_size > 200, "troubleshooting.md is suspiciously short"

def test_troubleshooting_covers_required_topics():
    content = TROUBLE.read_text().lower()
    for topic in REQUIRED_TROUBLESHOOTING_TOPICS:
        assert topic in content, f"Missing troubleshooting topic: {topic}"

def test_troubleshooting_entries_have_symptom_cause_resolution():
    content = TROUBLE.read_text()
    # Each ## heading should be followed by symptom/cause/resolution patterns
    sections = re.split(r"^##\s+", content, flags=re.M)[1:]
    for section in sections:
        lower = section.lower()
        assert any(k in lower for k in ["symptom", "issue", "problem"]), \
            f"Section missing symptom: {section.splitlines()[0]}"
        assert any(k in lower for k in ["cause", "reason", "why"]), \
            f"Section missing cause: {section.splitlines()[0]}"
        assert any(k in lower for k in ["resolution", "fix", "solution", "run", "check"]), \
            f"Section missing resolution: {section.splitlines()[0]}"

def test_deployment_covers_database_reverse_proxy_ssl():
    content = DEPLOY.read_text().lower()
    for section in ["database", "reverse proxy", "ssl"]:
        assert section in content, f"deployment.md missing section: {section}"

def test_deployment_has_nginx_and_traefik_examples():
    content = DEPLOY.read_text().lower()
    assert "nginx" in content and "server {" in content, "Missing Nginx example"
    assert "traefik" in content, "Missing Traefik example"

def test_no_placeholder_text():
    for f in [TROUBLE, DEPLOY]:
        content = f.read_text()
        for pat in PLACEHOLDER_PATTERNS:
            assert not re.search(pat, content), f"Placeholder {pat} found in {f.name}"

def test_code_fences_are_balanced():
    for f in [TROUBLE, DEPLOY]:
        content = f.read_text()
        fences = re.findall(r"^```", content, flags=re.M)
        assert len(fences) % 2 == 0, f"Unbalanced code fences in {f.name}"

def test_internal_links_resolve():
    for f in [TROUBLE, DEPLOY]:
        content = f.read_text()
        links = re.findall(r"\[([^\]]+)\]\(([^)]+)\)", content)
        for label, href in links:
            if href.startswith(("http://", "https://", "#", "mailto:")):
                continue
            target = (DOCS / href.split("#")[0]).resolve()
            assert target.exists(), f"Broken link in {f.name}: {href}"
