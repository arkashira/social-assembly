```javascript
// src/server.js
const WebSocket = require('ws');
const express = require('express');
const http = require('http');

const app = express();
const server = http.createServer(app);
const wss = new WebSocket.Server({ server });

wss.on('connection', (ws) => {
    console.log('New client connected');
    
    ws.on('message', (message) => {
        console.log(`Received: ${message}`);
        // Broadcast to all clients
        wss.clients.forEach((client) => {
            if (client.readyState === WebSocket.OPEN) {
                client.send(message);
            }
        });
    });

    ws.on('close', () => {
        console.log('Client disconnected');
    });
});

server.listen(3000, () => {
    console.log('Server is listening on port 3000');
});
```

## 1. Acceptance Criteria
1. **Create rule** – Moderator can submit a rule with *title* (≤100 chars), *description* (≤1000 chars), and *severity* (Low, Medium, High). The rule appears in the rule list immediately after creation.  
2. **Display rule** – All active rules are shown in a dedicated “Community Rules” banner on every forum page, ordered by severity (High → Low) and then creation date (newest first).  
3. **Edit rule** – Moderator can open an edit dialog, modify any field, and save; the list updates and a new version entry is created.  
4. **Archive rule** – Moderator can archive a rule; archived rules disappear from the public banner but remain visible in the moderator view with an “Archived” label.  
5. **Versioning & timestamps** – Each rule record stores `version` (incremental integer) and `createdAt` / `updatedAt` ISO‑8601 timestamps. The UI shows the current version and the “last updated” time for moderators.  
6. **Permission enforcement** – Only users with the `moderator` role see the create/edit/archive controls; regular users see only the read‑only banner.  
7. **Responsive & accessible** – UI passes WCAG 2.1 AA contrast checks and renders correctly on viewport widths ≥320 px and ≤1920 px.
