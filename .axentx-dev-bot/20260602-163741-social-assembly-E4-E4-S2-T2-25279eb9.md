const crypto = require('crypto');

/**
 * Generates a unique token for a community.
 * @param {string} communityName - The name of the community.
 * @returns {Promise<string>} A 64-character hexadecimal token.
 */
const generateToken = (communityName) => {
  const token = crypto.randomBytes(32).toString('hex');
  return token;
};

/**
 * Constructs the full invite URL with the pre-filled community name.
 * @param {string} communityName - The community name.
 * @param {string} token - The unique token.
 * @returns {string} The complete invite URL.
 */
const getInviteUrl = (communityName, token) => {
  return `https://example.com/join?community=${encodeURIComponent(communityName)}&token=${token}`;
};

module.exports = {
  generateToken,
  getInviteUrl,
};
```

        setData(response.data);
      } catch (error) {
        setError(error);

};
