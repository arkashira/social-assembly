```javascript
import { mount } from '@vue/test-utils';
import Thread from '@/components/Thread.vue';
import { createStore } from 'vuex';

describe('Thread.vue', () => {
  let store;
  let actions;
  let state;

  beforeEach(() => {
    actions = {
      fetchThread: jest.fn(),
      createPost: jest.fn(),
      updatePost: jest.fn(),
      deletePost: jest.fn(),
      createReply: jest.fn(),
    };
    state = {
      thread: {
        id: '1',
        title: 'Test Thread',
        posts: [
          { id: '1', userId: 'user1', content: '# Header', createdAt: '2023-01-01' },
          { id: '2', userId: 'user2', content: '- List item', createdAt: '2023-01-02' },
        ],
        hasThreadAccess: true,
      },
      currentUserId: 'user1',
      postsPerPage: 20,
      replyLimit: 3,
    };
    store = createStore({
      state,
      actions,
      getters: {
        hasThreadAccess: () => state.thread.hasThreadAccess,
      },
    });
  });

  it('renders thread title and header', () => {
    const wrapper = mount(Thread, { global: { plugins: [store] } });
    expect(wrapper.find('.thread-header h1').text()).toBe('Test Thread');
  });

  it('displays posts sorted by createdAt descending', () => {
    const wrapper = mount(Thread, { global: { plugins: [store] } });
    const posts = wrapper.findAll('.post');
    expect(posts[0].text()).toContain('2023-01-02'); // Newer post first
  });

  it('shows edit/delete buttons only for owned posts', async () => {
    const wrapper = mount(Thread, { global: { plugins: [store] } });
    const ownedPost = wrapper.findAll('.post').at(0);
    const unownedPost = wrapper.findAll('.post').at(1);
    expect(ownedPost.find('.edit-button').exists()).toBe(true);
    expect(unownedPost.find('.edit-button').exists()).toBe(false);
  });

  it('toggles reply form when replyTarget is set', async () => {
    const wrapper = mount(Thread, { global: { plugins: [store] } });
    await wrapper.find('.reply-button').trigger('click');
    expect(wrapper.find('.reply-form').exists()).toBe(true);
  });

  it('calls createPost with correct payload when submitting new post', async () => {
    const wrapper = mount(Thread, { global: { plugins: [store] } });
    await wrapper.find('.new-post-form textarea').setValue('New post');
    await wrapper.find('.new-post-form button').trigger('click');
    expect(actions.createPost).toHaveBeenCalledWith(expect.anything(), {
      threadId: '1',
      content: 'New post',
    });
  });

  it('shows error message on failed API call', async () => {
    actions.createPost.mockRejectedValue(new Error('Failed'));
    const wrapper = mount(Thread, { global: { plugins: [store] } });
    await wrapper.find('.new-post-form textarea').setValue('New post');
    await wrapper.find('.new-post-form button').trigger('click');
    expect(wrapper.find('.error-message').text()).toContain('Failed');
  });
});
```
