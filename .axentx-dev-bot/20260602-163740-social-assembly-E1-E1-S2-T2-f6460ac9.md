+import axios from 'axios';
+
+/**
+ * PostEditor – reusable editor for creating or updating a post.
+ *
+ * Props
+ * -----
+ * post        – optional object { id?, title?, content? }.
+ * onSuccess   – callback invoked with the saved post after a successful mutation.
+ * onCancel    – callback invoked when the user clicks the Cancel button.

+  const [content, setContent] = useState(post.content || '');
+
+  // -------------------------------------------------------------------------
+  // React‑Query helpers
+  // -------------------------------------------------------------------------
+  const queryClient = useQueryClient();
+
+  // Mutation: POST for new, PUT for existing
+  const saveMutation = useMutation(
+    async (payload) => {
+      if (post.id) {
+        const { data } = await axios.put(`/api/posts/${post.id}`, payload);
+        return data;
+      }
+      const { data } = await axios.post('/api/posts', payload);
+      return data;
+    },
+    {
+      onSuccess: (savedPost) => {
+        queryClient.invalidateQueries(['posts']);
+        if (onSuccess) onSuccess(savedPost);
+      },
+      onError: (error) => {
+        console.error('Failed to save post:', error);
+        alert('Unable to save the post. See console for details.');
+      },
+    }
+  );
+
+  // -------------------------------------------------------------------------
+  // Sync local state when the incoming `post` prop changes
