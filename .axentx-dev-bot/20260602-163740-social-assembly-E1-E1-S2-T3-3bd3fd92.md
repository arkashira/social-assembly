const { Model, DataTypes } = require('sequelize');
const sequelize = require('../config/database');

class Post extends Model {}

Post.init({
  title: {
    type: DataTypes.STRING,
    allowNull: false
  },
  body: {
    type: DataTypes.TEXT,
    allowNull: false
  },
  mediaUrl: {
    type: DataTypes.STRING,
    allowNull: true
  },

    type: DataTypes.INTEGER,
    allowNull: false
  },
+  // Soft‑delete marker
+  deletedAt: {
+    type: DataTypes.DATE,
+    allowNull: true,
+    defaultValue: null
+  }
}, {
  sequelize,
  modelName: 'Post',
  tableName: 'posts',
  timestamps: true,
  // Exclude soft‑deleted rows by default
+  defaultScope: {
+    where: {
+      deletedAt: null
+    }
+  },
+  scopes: {
+    // Include deleted rows when explicitly requested
+    withDeleted: {}
+  }
});

module.exports = Post;
```

### 2. Add DELETE route

```diff
// /opt/axentx/social-assembly/backend/routes/post.js
const express = require('express');
const router = express.Router();
const Post = require('../models/Post');
const { authenticate } = require('../middleware/auth');

// Existing routes (create, read, update) omitted for brevity

/**
 * @route   DELETE /posts/:id
 * @desc    Soft‑delete a post
 * @access  Author or moderator
 */
router.delete('/:id', authenticate, async (req, res) => {
  try {
    const post = await Post.findByPk(req.params.id);
    if (!post) {
      return res.status(404).json({ error: 'Post not found' });
    }

    // Already deleted
    if (post.deletedAt) {
      return res.status(410).json({ error: 'Post already deleted' });
    }

    // Permission check
    const isAuthor = post.authorId === req.user.id;
    const isModerator = req.user.role === 'moderator';
    if (!isAuthor && !isModerator) {
      return res.status(403).json({ error: 'Forbidden' });
    }

    await post.update({ deletedAt: new Date() });

    return res.status(200).json({ message: 'Post deleted successfully' });
  } catch (err) {
    console.error(err);
    return res.status(500).json({ error: 'Server error' });
  }
});

module.exports = router;
```

> **Note**:  
> *`authenticate` middleware* is assumed to populate `req.user` with `{ id, role }`.  
> The default scope in the model ensures that all standard queries (`Post.findAll`, `Post.findByPk` without a scope) automatically exclude soft‑deleted posts.

### 3. (Optional) Update existing GET routes to use the default scope
If any route explicitly queried all posts without a scope, it will now automatically exclude deleted ones. No changes required unless a route intentionally used `Post.scope('withDeleted')`.
