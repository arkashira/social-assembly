// /opt/axentx/social-assembly/backend/routes/post.js

    const post = await Post.findById(req.params.id);

      return res.status(404).json({ message: 'Post not found' });

    // Check if user is author or moderator
    if (req.user.id !== post.author && !req.user.isModerator) {
      return res.status(403).json({ message: 'Forbidden' });

    // Set deleted flag to true
    post.deleted = true;
    await post.save();
    res.status(200).json({ message: 'Post deleted successfully' });
  } catch (error) {
    res.status(500).json({ message: 'Error deleting post', error });
  }
});

// Update the get method to only retrieve non-deleted posts
router.get('/', async (req, res) => {
  const posts = await Post.find({ deleted: false });
  res.status(200).json(posts);
});

module.exports = router;

## 1. Acceptance Criteria (measurable)

| # | Criterion | Measurement |
|---|-----------|-------------|
| 1 | **Soft delete flag** | `Post.isDeleted` (boolean) is set to `true` when a delete request is processed. |
| 2 | **Author or moderator only** | Only the author (`post.authorId`) or a user with role `moderator` can set `isDeleted`. |
| 3 | **Timestamp recorded** | `post.deletedAt` (ISO timestamp) is populated on soft delete. |
| 4 | **Post visibility** | Soft‑deleted posts are excluded from standard list/query endpoints (`GET /posts`). |
| 5 | **Hard delete protection** | No hard delete endpoint exists; `DELETE /posts/:id` performs soft delete only. |
| 6 | **History preservation** | Deleted posts can be restored (`PUT /posts/:id/restore`) by the author or moderator, resetting `isDeleted` and clearing `deletedAt`. |
| 7 | **Audit trail** | Each soft delete/restore action logs an entry in `PostHistory` with `action`, `userId`, and `timestamp`. |

## 2. Unit Tests (pseudo‑code)

```js
// tests/unit/PostModel.test.js
const { Post, User, PostHistory } = require('../../models');

describe('Post Model Soft Delete', () => {
  let post, author, moderator, otherUser;

  beforeEach(async () => {
    author = await User.create({ username: 'author', role: 'member' });
    moderator = await User.create({ username: 'mod', role: 'moderator' });
    otherUser = await User.create({ username: 'intruder', role: 'member' });
    post = await Post.create({
      title: 'Test',
      body: 'Content',
      authorId: author.id,
    });
  });

  test('author can soft delete within 5 minutes', async () => {
    await post.softDelete(author.id);
    const updated = await Post.findByPk(post.id);
    expect(updated.isDeleted).toBe(true);
    expect(updated.deletedAt).toBeInstanceOf(Date);
  });

  test('moderator can soft delete any post', async () => {
    await post.softDelete(moderator.id);
    const updated = await Post.findByPk(post.id);
    expect(updated.isDeleted).toBe(true);
  });

  test('non-author/non-moderator cannot delete', async () => {
    await expect(post.softDelete(otherUser.id))
      .rejects
      .toThrow('Permission denied');
  });

  test('soft delete records history', async () => {
    await post.softDelete(author.id);
    const history = await PostHistory.findOne({ where: { postId: post.id } });
    expect(history.action).toBe('soft_delete');
    expect(history.userId).toBe(author.id);
  });

  test('restore resets flags and logs history', async () => {
    await post.softDelete(author.id);
    await post.restore(author.id);
    const updated = await Post.findByPk(post.id);
    expect(updated.isDeleted).toBe(false);
    expect(updated.deletedAt).toBeNull();

    const restoreHist = await PostHistory.findOne({
      where: { postId: post.id, action: 'restore' }
    });
    expect(restoreHist).not.toBeNull();
  });
});
```
