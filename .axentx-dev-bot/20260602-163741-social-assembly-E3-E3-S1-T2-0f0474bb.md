### 2. `src/lib/content-filter.ts`

```ts
/**
 * Lightweight bias/tell filter inspired by the stop‑slop portfolio.
 * Returns a risk score (0–1) and an array of flagged phrases.
 */
export interface FilterResult {
  risk: number;          // 0 = safe, 1 = high risk
  flags: string[];       // phrases that triggered the risk
}

const STOP_WORDS = [
  // Common bias/tell patterns
  /hate/i,
  /violence/i,
  /sexist/i,
  /racist/i,
  /misogynist/i,
  /bigot/i,
  /slur/i,
  /harassment/i,
];

export function analyzeContent(text: string): FilterResult {
  const flags: string[] = [];
  STOP_WORDS.forEach((regex) => {
    const matches = text.match(regex);
    if (matches) flags.push(matches[0]);
  });

  const risk = flags.length / STOP_WORDS.length;
  return { risk, flags };
}
```

```ts
import { Router } from 'express';
import { v4 as uuidv4 } from 'uuid';
import { analyzeContent, FilterResult } from '../../lib/content-filter';
import { Post } from '../../types';

const router = Router();

/**
 * POST /api/posts
 * Body: { authorId: string, text: string }
 */
router.post('/', (req, res) => {
  const { authorId, text } = req.body;
  if (!authorId || !text) {
    return res.status(400).json({ error: 'authorId and text required' });
  }

  const risk: FilterResult = analyzeContent(text);

  const post: Post = {
    id: uuidv4(),
    authorId,
    text,
    createdAt: new Date().toISOString(),
    risk,
  };
```
