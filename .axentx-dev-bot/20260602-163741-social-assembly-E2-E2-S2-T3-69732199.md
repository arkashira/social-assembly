/**
 * Sends a notification to moderators when a new report is submitted.
 * @param {Object} report - The report object containing reporter info and timestamp.
 */
async function sendModeratorNotification(report) {
  const transporter = nodemailer.createTransport({
    host: 'smtp.example.com',
    port: 587,
    secure: false, // or 'STARTTLS'
    auth: {
      user: 'username',
      pass: 'password',
    },
  });

  const mailOptions = {
    from: 'report@example.com',
    to: 'moderators@example.com',
    subject: 'New Report Submitted',
    text: `A new report has been submitted by ${report.reporter.username} at ${report.timestamp}. Reason: ${report.reason}. Context: ${report.context}`,
  };

  try {
    await transporter.sendMail(mailOptions);
    console.log('Moderator notification sent successfully');
  } catch (error) {
    console.error('Error sending moderator notification:', error);
  }
}

module.exports = { sendModeratorNotification };
