	"bytes"

	"gopkg.in/yaml.v3"

// -----------------------------------------------------------------------------
// 1.  YAML schema
// -----------------------------------------------------------------------------

// ActionType enumerates the possible moderation actions.
type ActionType string

const (
	HidePost ActionType = "hide_post"
	TempMute ActionType = "temp_mute"
	PermBan  ActionType = "perm_ban"
)

// Action represents a single moderation rule that can be triggered.
type Action struct {
	Type   ActionType `yaml:"type"`   // required
	Reason string     `yaml:"reason"` // required

	// Optional parameters – only used for TempMute
	Duration time.Duration `yaml:"duration,omitempty"`
}

// RateLimitRule defines a simple rate‑limiting policy.
type RateLimitRule struct {
	PostsPerHour int `yaml:"posts_per_hour"`
}

// UserBanRule defines a ban rule for a specific user or pattern.
type UserBanRule struct {
	Username string `yaml:"username"`
	Reason   string `yaml:"reason"`
}

// RuleSet is the top‑level structure that the YAML file must match.
type RuleSet struct {
	WordFilter []string          `yaml:"word_filter"` // list of banned words/phrases
	RateLimit  *RateLimitRule    `yaml:"rate_limit"`
	UserBans   []UserBanRule     `yaml:"user_bans"`
	Actions    []Action          `yaml:"actions"` // global actions that apply to any post
}

// -----------------------------------------------------------------------------
// 2.  Parser – thread‑safe, reloadable
// -----------------------------------------------------------------------------

// Parser holds the current rule set and a mutex for concurrent access.
type Parser struct {
	mu    sync.RWMutex
	rules *RuleSet
	path  string // keep the file path for reload
}

// NewParser creates a new Parser instance and optionally loads a file.
func NewParser(path string) (*Parser, error) {
	p := &Parser{path: path}
	if path != "" {
		if err := p.LoadFromFile(path); err != nil {
			return nil, err
		}
	}
	return p, nil
}

		return fmt.Errorf("cannot read rule file %q: %w", path, err)

	if len(bytes.TrimSpace(data)) == 0 {

func TestNewParser(t *testing.T) {
	p := NewParser()
	assert.NotNil(t, p)
	assert.NotNil(t, p.rules)
	assert.NotNil(t, p.mu)
}

func TestLoadFromFile_ValidYAML(t *testing.T) {
	tmpFile := writeTempYAML(t, `
word_filters:
  - "spam"
rate_limits:
  posts_per_hour: 10
bans:
  - user_id: 123
`)
	defer os.Remove(tmpFile)

	p := NewParser()
	err := p.LoadFromFile(tmpFile)
	require.NoError(t, err)
	assert.Len(t, p.rules.WordFilters, 1)
	assert.Equal(t, 10, p.rules.RateLimits.PostsPerHour)
	assert.Len(t, p.rules.Bans, 1)
}

func TestLoadFromFile_UnknownTopLevelKey(t *testing.T) {
	tmpFile := writeTempYAML(t, `
unknown_field: true

`)
	defer os.Remove(tmpFile)

	p := NewParser()
	err := p.LoadFromFile(tmpFile)
	require.Error(t, err)
	assert.Contains(t, err.Error(), "unknown_field")
}

func TestLoadFromFile_EmptyFile(t *testing.T) {

	err := p.LoadFromFile(tmpFile)

func TestGetRules_DeepCopy(t *testing.T) {
	tmpFile := writeTempYAML(t, `
word_filters:
  - "test"
`)
	defer os.Remove(tmpFile)

	p := NewParser()
	require.NoError(t, p.LoadFromFile(tmpFile))

	rules := p.GetRules()
	rules.WordFilters[0] = "mutated"

	// Internal state should remain unchanged
	internalRules := p.GetRules()
	assert.Equal(t, "test", internalRules.WordFilters[0])
}

func TestReload_ConcurrentWithGetRules(t *testing.T) {
	tmpFile := writeTempYAML(t, `
word_filters:
  - "v1"
`)
	defer os.Remove(tmpFile)

	p := NewParser()
	require.NoError(t, p.LoadFromFile(tmpFile))

	var wg sync.WaitGroup
	for i := 0; i < 10; i++ {
		wg.Add(2)
		go func() {
			defer wg.Done()
			_ = p.GetRules()
		}()
		go func(v int) {
			defer wg.Done()
			newContent := fmt.Sprintf("word_filters:\n  - \"v%d\"\n", v)
			writeFile(t, tmpFile, newContent)
			p.Reload()
		}(i)
	}
	wg.Wait()

	// Final state should reflect last write
	rules := p.GetRules()
	assert.Equal(t, "v9", rules.WordFilters[0])
}

func writeTempYAML(t *testing.T, content string) string {
	t.Helper()
	f, err := os.CreateTemp("", "rules*.yaml")
	require.NoError(t, err)
	_, err = f.WriteString(content)
	require.NoError(t, err)
	f.Close()
	return f.Name()
}

func writeFile(t *testing.T, path, content string) {
	t.Helper()
	require.NoError(t, os.WriteFile(path, []byte(content), 0644))
}
